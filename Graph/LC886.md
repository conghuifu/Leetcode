### Solution 
```
class Solution:
    def possibleBipartition(self, n: int, dislikes: List[List[int]]) -> bool:
        # build graph
        graph = collections.defaultdict(set)
        for s, e in dislikes:
            graph[s].add(e)
            graph[e].add(s)
        
        # build visited
        visited = [-1 for i in range(n+1)]
        
        # bipartition
        for i in range(n):
            if visited[i] != -1:
                continue
            else:
                queue = collections.deque([(i, 0)])
                
                while queue:
                    a, group = queue.popleft()
                    for b in graph[a]:
                        if visited[b] != -1:
                            if visited[b] == group:
                                return False
                        else:
                            visited[b] = 1-group
                            queue.append((b, 1-group))
                            
        return True
```
