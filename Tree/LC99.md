### Solution
1. 记住如何找到两个异常值 2. 第二个找不到的特殊情况是直接跟在第一个后面
1. stack
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        res = []
        self.preorder(root, res)
        
        n = len(res)
        first = None
        sec = None
        curMax = TreeNode(-sys.maxsize)
        
        for i in range(n):
            if curMax.val <= res[i].val:
                curMax = res[i]
            else:
                if not first:
                    first = curMax
                    curMax = res[i]
                    sec = res[i]
                else:
                    sec = res[i]
        first.val, sec.val = sec.val, first.val
        return root
        
        
    def preorder(self, root, res):
        if not root:
            return
        self.preorder(root.left, res)
        res.append(root)
        self.preorder(root.right, res)
```
2. directly inorder
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        self.curMax = TreeNode(-sys.maxsize)
        self.first = None
        self.sec = None
        self.preorder(root)
        
        self.first.val, self.sec.val = self.sec.val, self.first.val
        return root
        
    def preorder(self, root):
        if not root:
            return
        
        self.preorder(root.left)
        
        if root.val >= self.curMax.val:
            self.curMax = root
        else:
            if not self.first:
                self.first = self.curMax
                self.sec = root
                self.curMax = root
            else:
                self.sec = root
        
        self.preorder(root.right)
```
