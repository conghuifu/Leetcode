### Solution
这题和285有点不一样，285只要判断一个方向，这题需要两个方向。所以recursive的方法，需要一个全局变量去记录最小，（不能直接返回），并且多写一个helper function. <br />
1. iterative 
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def closestValue(self, root: Optional[TreeNode], target: float) -> int:
        if not root: return None
        
        diff = sys.maxsize
        res = None
        
        while root:
            if abs(root.val - target) < diff:
                    diff = abs(root.val - target)
                    res = root.val
                    
            if root.val == target:
                return root.val
            if root.val > target:
                root = root.left
            else:
                root = root.right
                
        return res
```
2. recursive
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def closestValue(self, root: Optional[TreeNode], target: float) -> int:
        self.closest = sys.maxsize
        self.dfs(root, target)
        return self.closest
        
        
        
    def dfs(self, root, target):
        if not root: return
        
        if abs(root.val - target) < abs(self.closest - target):
            self.closest = root.val
        
        if root.val > target:
            self.dfs(root.left, target)
        else:
            self.dfs(root.right, target)
```
3. recursive <br />
类似285的做法
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def closestValue(self, root: Optional[TreeNode], target: float) -> int:
        upper = self.largerOrEqual(root, target)
        lower = self.smallerOrEqual(root, target)
        
        if upper and lower:
            if abs(upper.val - target) > abs(lower.val - target):
                return lower.val
            else:
                return upper.val
        elif upper:
            return upper.val
        elif lower:
            return lower.val
        else:
            return None
        
        
    def smallerOrEqual(self, root, target):
        if not root: return None
        
        if root.val > target:
            return self.smallerOrEqual(root.left, target)
        else:
            return self.smallerOrEqual(root.right, target) or root
        
    def largerOrEqual(self, root, target):
        if not root: return None
        
        if root.val < target:
            return self.largerOrEqual(root.right, target)
        else:
            return self.largerOrEqual(root.left, target) or root
```
