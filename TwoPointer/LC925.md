#### Solution 925 Two Pointers
1. Easy understand
```
class Solution:
    def isLongPressedName(self, name: str, typed: str) -> bool:
        m = len(name)
        n = len(typed)
        x = 0
        y = 0
        
        while (x<m) or (y < n):
            if x < m:
                xchar = name[x]
            if y < n:
                ychar = typed[y]
            if xchar == ychar:
                x += 1
                y += 1
            else:
                if y >= n: return False
                ychar, y = self.nextChar(typed, y)
                if xchar == ychar:
                    y += 1
                    x += 1
                else:
                    return False
        return True
    
    def nextChar(self, typed, y):
        char = typed[y]
        while (y > 0) and (y < len(typed)):
            if typed[y] != typed[y - 1]:
                char = typed[y]
                return char, y
            else:
                y += 1
        return char, y
```

class Solution:
    def isLongPressedName(self, name: str, typed: str) -> bool:
        m = len(name)
        n = len(typed)
        
        i, j = 0, 0
        while j < n:
            if (i < m) and (name[i] == typed[j]):
                i += 1
                j += 1
            else:
                if (j > 0) and (typed[j] == typed[j - 1]):
                    j += 1
                else:
                    return False
        
        return i == m
