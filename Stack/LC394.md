```
class Solution:
    def decodeString(self, s: str) -> str:
        stack = collections.deque()
        res = ''
        num = '0'
        
        for i in s:
            if i.isdigit():
                num += i
            elif i not in '[]':
                num = max(int(num), 1)
                res += num*i
                num = '0'
            elif i == '[':
                stack.append([num, res])
                num = '0'
                res = ''
            elif i == ']':
                lastNum, lastRes = stack.pop()
                res = lastRes + max(int(lastNum), 1) * res
                num = '0'
        
        return res
```
