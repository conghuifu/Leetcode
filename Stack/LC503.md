class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        stack = collections.deque()
        n = len(nums)
        res = [-1 for i in range(n)]
        
        for i in [i for i in range(n)] * 2:
            while stack and (nums[stack[-1]] < nums[i]):
                res[stack.pop()] = nums[i]
            stack.append(i)
        return res
