### Solution
1. O(n) space, O(n) time
```
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        # first round
        p = 1
        output = []
        for i in range(n):
            output.append(p)
            p *= nums[i]
            
        # reverse
        p = 1
        for i in range(n-1, -1, -1):
            output[i] *= p
            p *= nums[i]
        return output
```
