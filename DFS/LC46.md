### Solution
backtrack. to prevent the situations like '111, 112, ....', judge if nums[i] in cur.
```
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        
        def backtrack(nums, cur):
            # basic cases
            if len(cur) == len(nums):
                res.append(cur[:])
                return
            
            # recursion
            for i in range(len(nums)):
                if nums[i] in cur:
                    continue
                cur.append(nums[i])
                backtrack(nums, cur)
                cur.pop()
                
        backtrack(nums, [])
        return res

```
or 
```
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        
        def backtrack(nums, cur, used):
            # basic case
            if len(cur) == len(nums):
                res.append(cur[:])
                return
            
            for i in range(len(nums)):
                if i in used:
                    continue
                cur.append(nums[i])
                used.add(i)
                backtrack(nums, cur, used)
                cur.pop()
                used.remove(i)
        
        backtrack(nums, [], set())
        return res
```
